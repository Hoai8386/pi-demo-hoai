Xác thực người dùng sau khi init xong Pi.authenticate(['payments', 'username'], { onIncompletePaymentFound: (payment) => { console.log("Có payment dang dở:", payment.paymentId); return payment; } }).then(auth => { console.log("Login Pi thành công!", auth.user); alert("Đã login Pi: " + auth.user.username); }).catch(err => { console.error("Lỗi authenticate:", err); alert("Lỗi login Pi: " + (err.message || err)); }); }).catch(err => { console.error("Lỗi Pi.init():", err); alert("Lỗi khởi tạo Pi SDK: " + err.message); }); // Hàm test thanh toán function testPayment() { const paymentData = { amount: 1, memo: "Test thanh toán demo 1 Pi", metadata: { type: "demo", item: "test_payment" } }; const callbacks = { onReadyForServerApproval: (paymentId) => { console.log("Payment ready:", paymentId); }, onReadyForServerCompletion: (paymentId, txid) => { console.log("Thanh toán thành công! TxID:", txid); alert("✅ Thanh toán thành công!\nTxID: " + txid); }, onCancel: (paymentId) => { alert("Bạn đã hủy thanh toán."); }, onError: (error) => { console.error(error); alert("Lỗi thanh toán: " + error.message); } }; Pi.createPayment(paymentData, callbacks); }
